# Script
A parser/interpreter for a simple script language

![Example](https://raw.githubusercontent.com/kristofferkjeldby/Script/main/logo.png)

#### Hello world
```
// Hello world
echoln("Hello world");
```
*Output*
```
Hello world
Script ran: 0.00 seconds
```

#### Variables
```
// Variable scope
set("global", "1");
{
  set("local", "2");
  echoln(isset("global"));
  echoln(isset("local"));
}
echoln(isset("global"));
echoln(isset("local"));
```
*Output*
```
True
True
True
False
Script ran: 0.01 seconds
```
#### Strings
```
// Strings
set("a", "part1");
set("b", "part2");
echoln(concat(get("a"), get("b")));
// String repeat
echoln(rep("a", "3"));
// String equality
echoln(eqs("1", "2"));
```
*Output*
```
part1part2
aaa
False
Script ran: 0.00 seconds
```

#### Math

// Math
echoln(add("1", "2"));
echoln(sub("1", "2"));
echoln(mul("1", "2"));

```
*Output*

0.5

```

// Logic

// Conditionals
if (eq("2", add("1", "1")))

#### Loops
```
// Loops
set("c", "1");
while (lt(get("c"), "4"))
{
  echoln(concat("Loop ", get("c")));
  set("c", add(get("c"), "1"));
}
```
*Output*
```
Loop 1
Loop 2
Loop 3
Script ran: 0.00 seconds
```

#### Label/Goto
```
// Labels/Goto
set("d", "1");
label 1;
echoln(concat("Goto ", get("d")));
set("d", add(get("d"), "1"));
if (lt(get("d"), "4"))
{
  goto 1;
}
```
*Output*
```
Goto 1
Goto 2
Goto 3
Script ran: 0.04 seconds
```

#### Mandelbrot
```
// The commonly shown part of the set
set("minx", "-2.1");
set("miny", "-1.3");
set("maxx", "1");
set("maxy", "1.3");

// Convert to a step size
set("xstep", div(sub(get("maxx"), get("minx")), "100"));
set("ystep", div(sub(get("maxy"), get("miny")), "45"));

set("x", get("minx"));
set("y", get("miny"));

// The limit used in the before discarding a coordinate
set("maxn", "15");

// Iterate y axis
while (lt(get("y"), get("maxy")))
{
  // Iterate x axis
  while (lt(get("x"), get("maxx")))
  {
    set("n", "0");
    set("c", " ");
    
    set("z", "0");
    set("zi", "0");
    
    // Determine whether coordinate is part of the set
    while (and(not(eqs(get("c"), "X")), lt(get("n"), get("maxn"))))
    {
      set("len", sqrt(add(sq(get("z")), sq(get("zi")))));
  
      // Not part of the set 
      if(gt(get("len"), "2"))
      {
        set("c", "X");
      }
      
      // Square a complex number
      set("zs", sub(sq(get("z")), sq(get("zi"))));
      set("zis", mul(mul(get("z"), get("zi")), "2"));

      set("z", add(get("zs"), get("x")));
      set("zi", add(get("zis"), get("y")));
      
      set("n", add(get("n"), "1"));
    }
    
    echo(get("c"));
    set("x", add(get("x"), get("xstep")));
  }
  echoln("");

  set("x", get("minx"));
  set("y", add(get("y"), get("ystep")));
}
```
